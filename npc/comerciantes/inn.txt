//======================Lighthalzen =================================
lhz_in02,230,284,4	script	Funcionário do Hotel#01	4_M_04,{
	mes "[Funcionário do Hotel]";
	mes "Dragão Real,";
	mes "onde você encontra as";
	mes "melhores acomodações";
	mes "e o melhor serviço.";
	next;
	switch(select("Salvar:Descansar - 5.000 zenys:Cancelar")) {
	case 1:
		savepoint "lhz_in02",209,275;
		mes "[Funcionário do Hotel]";
		mes "Obrigado, seu";
		mes "Ponto de Retorno foi";
		mes "salvo aqui no";
		mes "Dragão Real.";
		close;
	case 2:
		if(Zeny < 5000) {
			mes "[Funcionário do Hotel]";
			mes "Desculpe, mas";
			mes "você precisa de 5.000 zenys";
			mes "para se hospedar aqui.";
			close;
		}
		set Zeny,Zeny - 5000;
		mes "[Funcionário do Hotel]";
		mes "Obrigado!";
		mes "Espero que goste";
		mes "de sua estadia no";
		mes "Dragão Real.";
		close2;
		warp "lhz_in02",219,150;
		percentheal 100,100;
		npcskill "AL_BLESSING",10,99,99;
		end;
	case 3:
		mes "[Funcionário do Hotel]";
		mes "Obrigado e";
		mes "tenha um bom dia.";
		close;
	}
}

//=======================Inn Function ==============================
function	script	F_InnMaid	{
	mes getarg(0);
	mes "Olá, você chegou à";
	mes getarg(1) + ".";
	mes "Em que posso ajudar?";
	next;
	switch(select("Salvar:Descansar - 5.000 zenys:Cancelar")) {
	case 1:
		mes getarg(0);
		mes "O seu Ponto de Retorno foi salvo.";
		mes "Obrigada, volte sempre.";
		savepoint getarg(2),getarg(3),getarg(4);
		close;
	case 2:
		mes getarg(0);
		if(Zeny < 5000){
			mes "Desculpe, a taxa de serviço é de 5.000 zenys.";
			mes "Por favor, tenha certeza de que tem dinheiro suficiente para esse serviço.";
			close;
		}
		mes "Obrigada.";
		mes "Espero que você tenha um bom descanso.";
		close2;
		set Zeny,Zeny - 5000;
		percentheal 100,100;
		return;
	case 3:
		close;
	}
}
